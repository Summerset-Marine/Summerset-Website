```javascript
import { Client } from '@hubspot/api-client'

function getClient() {
  return new Client({ accessToken: process.env.HUBSPOT_ACCESS_TOKEN })
}

// ── HubSpot Forms API submission ──────────────────────────────────────────────
// Submits to HubSpot Forms API v3 (no auth required on this endpoint —
// uses portal ID and form ID only). This is the simplest, most reliable method.
async function submitToHubSpotForm(formId, fields, context = {}) {
  const portalId = process.env.HUBSPOT_PORTAL_ID

  const body = {
    fields: Object.entries(fields).map(([name, value]) => ({
      objectTypeId: '0-1', // Contact object
      name,
      value: value ?? '',
    })),
    context: {
      pageUri: 'https://summersetmarine.com',
      pageName: 'Summerset Marine Construction',
      ...context,
    },
  }

  const response = await fetch(
    `https://api.hsforms.com/submissions/v3/integration/submit/${portalId}/${formId}`,
    {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(body),
    }
  )

  if (!response.ok) {
    const error = await response.text()
    throw new Error(`HubSpot Forms API error: ${response.status} — ${error}`)
  }

  return response.json()
}

// ── Create a Deal in HubSpot CRM for consultation requests ───────────────────
async function createDeal(contactEmail, data) {
  const client = getClient()

  // First find or confirm the contact was created by the form submission
  // Then create a deal associated with that contact
  const dealResponse = await client.crm.deals.basicApi.create({
    properties: {
      dealname: `Consultation Request — ${data.name} — ${data.lake || data.market || 'General'}`,
      pipeline: 'default',
      dealstage: 'appointmentscheduled', // First stage — adjust to SMC's actual pipeline stage ID
      amount: '',
      closedate: '',
      description: data.projectDescription || '',
      hs_note_status: `
        Name: ${data.name}
        Email: ${data.email}
        Phone: ${data.phone || 'Not provided'}
        Lake / Location: ${data.lake || 'Not provided'}
        Market: ${data.market || 'Not provided'}
        Project Description: ${data.projectDescription || 'Not provided'}
        Timeframe: ${data.timeframe || 'Not provided'}
        How they heard about SMC: ${data.hearAbout || 'Not provided'}
      `.trim(),
    },
  })

  // Associate deal with contact by email
  try {
    const contacts = await client.crm.contacts.searchApi.doSearch({
      filterGroups: [{
        filters: [{ propertyName: 'email', operator: 'EQ', value: contactEmail }]
      }],
      properties: ['email'],
      limit: 1,
    })

    if (contacts.results.length > 0) {
      const contactId = contacts.results[0].id
      await client.crm.deals.associationsApi.create(
        dealResponse.id,
        'contacts',
        contactId,
        [{ associationCategory: 'HUBSPOT_DEFINED', associationTypeId: 3 }]
      )
    }
  } catch (assocErr) {
    // Non-fatal — deal is created even if association fails
    console.warn('[HubSpot] Deal-contact association failed:', assocErr.message)
  }

  return dealResponse
}

// ── Public submission functions ───────────────────────────────────────────────

export async function submitConsultationForm(data, context) {
  const formId = process.env.HUBSPOT_FORM_ID_CONSULTATION

  // Map React form fields to HubSpot field names
  // NOTE: field names must match the field internal names in the HubSpot form exactly
  const fields = {
    firstname:        data.name?.split(' ')[0] || data.name,
    lastname:         data.name?.split(' ').slice(1).join(' ') || '',
    email:            data.email,
    phone:            data.phone || '',
    city:             data.lake || '',           // maps to lake/location field
    state:            data.market || '',         // maps to market field
    message:          data.projectDescription || '',
    hs_lead_status:   'NEW',
    // Custom HubSpot properties — TODO: replace with actual HubSpot custom property names
    // once SMC configures them in their portal
    smc_timeframe:    data.timeframe || '',
    smc_hear_about:   data.hearAbout || '',
  }

  await submitToHubSpotForm(formId, fields, context)

  // Create a Deal for consultation requests
  await createDeal(data.email, data)
}

export async function submitContactForm(data, context) {
  const formId = process.env.HUBSPOT_FORM_ID_CONTACT

  const fields = {
    firstname:  data.name?.split(' ')[0] || data.name,
    lastname:   data.name?.split(' ').slice(1).join(' ') || '',
    email:      data.email,
    phone:      data.phone || '',
    message:    data.message || '',
  }

  await submitToHubSpotForm(formId, fields, context)
  // Contact only — no Deal created for general contact submissions
}

export async function submitServiceRequestForm(data, context) {
  const formId = process.env.HUBSPOT_FORM_ID_SERVICE_REQUEST

  const fields = {
    firstname:          data.name?.split(' ')[0] || data.name,
    lastname:           data.name?.split(' ').slice(1).join(' ') || '',
    email:              data.email,
    phone:              data.phone || '',
    address:            data.propertyLocation || '',
    message:            data.issueDescription || '',
    // Custom HubSpot properties — TODO: replace with actual custom property names
    smc_service_type:   data.serviceType || '',
    smc_equipment:      data.equipmentDescription || '',
    smc_preferred_time: data.preferredTime || '',
  }

  await submitToHubSpotForm(formId, fields, context)
  // Contact only — no Deal created for service requests
}
```

REWRITE src/api/submit-form.js — UNIFIED FORM HANDLER (HUBSPOT):

Replace the entire existing file with this:

```javascript
import {
  submitConsultationForm,
  submitContactForm,
  submitServiceRequestForm,
} from '../lib/hubspot.js'

export default async function submitForm(req, res) {
  const { formType, ...data } = req.body

  // Basic validation
  if (!data.name || !data.email) {
    return res.status(400).json({ error: 'Name and email are required' })
  }

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/
  if (!emailRegex.test(data.email)) {
    return res.status(400).json({ error: 'Invalid email address' })
  }

  // Build HubSpot page context from request headers
  const context = {
    pageUri: `https://summersetmarine.com${req.headers.referer ? new URL(req.headers.referer).pathname : '/'}`,
    pageName: 'Summerset Marine Construction',
    ipAddress: req.headers['x-forwarded-for'] || req.socket.remoteAddress || '',
    hutk: req.cookies?.hubspotutk || '', // HubSpot tracking cookie — requires cookie-parser middleware
  }

  try {
    if (formType === 'consultation') {
      if (!data.projectDescription) {
        return res.status(400).json({ error: 'Project description is required' })
      }
      await submitConsultationForm(data, context)

    } else if (formType === 'contact') {
      if (!data.message) {
        return res.status(400).json({ error: 'Message is required' })
      }
      await submitContactForm(data, context)

    } else if (formType === 'service-request') {
      if (!data.issueDescription) {
        return res.status(400).json({ error: 'Issue description is required' })
      }
      await submitServiceRequestForm(data, context)

    } else {
      return res.status(400).json({ error: 'Invalid form type' })
    }

    res.status(200).json({ success: true })

  } catch (err) {
    console.error('[Form Submission] HubSpot submission failed:', err.message)
    res.status(500).json({ error: 'Form submission failed' })
  }
}
```
