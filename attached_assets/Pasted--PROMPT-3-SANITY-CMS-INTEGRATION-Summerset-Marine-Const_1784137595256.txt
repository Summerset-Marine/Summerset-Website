BUILD src/api/netsuite-webhook.js — WEBHOOK HANDLER:

This Express route receives POST requests from NetSuite when inventory changes. It validates the webhook secret, then updates src/data/inventory-cache.json with the new inventory data.

```javascript
import fs from 'fs'
import path from 'path'
import crypto from 'crypto'

const CACHE_PATH = path.join(process.cwd(), 'src/data/inventory-cache.json')

export default function netsuiteWebhook(req, res) {
  // Validate webhook secret
  const signature = req.headers['x-netsuite-signature']
  const expected = crypto
    .createHmac('sha256', process.env.NETSUITE_WEBHOOK_SECRET)
    .update(JSON.stringify(req.body))
    .digest('hex')

  if (signature !== expected) {
    return res.status(401).json({ error: 'Invalid webhook signature' })
  }

  // Write updated inventory to cache file
  const payload = {
    lastUpdated: new Date().toISOString(),
    items: req.body.items || [],
  }

  try {
    fs.writeFileSync(CACHE_PATH, JSON.stringify(payload, null, 2))
    console.log(`[NetSuite Webhook] Inventory cache updated — ${payload.items.length} items`)
    res.status(200).json({ success: true, itemCount: payload.items.length })
  } catch (err) {
    console.error('[NetSuite Webhook] Cache write failed:', err)
    res.status(500).json({ error: 'Cache write failed' })
  }
}
```

BUILD src/lib/netsuite.js — NETSUITE REST API CLIENT:

This module fetches inventory directly from NetSuite when needed (manual refresh or initial load if cache is empty). Uses OAuth 1.0a with TBA (Token-Based Authentication) as required by NetSuite REST API.

```javascript
import crypto from 'crypto'
import axios from 'axios'

function generateOAuthHeader(method, url) {
  const timestamp = Math.floor(Date.now() / 1000).toString()
  const nonce = crypto.randomBytes(16).toString('hex')

  const params = {
    oauth_consumer_key: process.env.NETSUITE_CONSUMER_KEY,
    oauth_nonce: nonce,
    oauth_signature_method: 'HMAC-SHA256',
    oauth_timestamp: timestamp,
    oauth_token: process.env.NETSUITE_TOKEN_ID,
    oauth_version: '1.0',
  }

  const baseString = [
    method.toUpperCase(),
    encodeURIComponent(url),
    encodeURIComponent(
      Object.keys(params).sort().map(k => `${k}=${params[k]}`).join('&')
    ),
  ].join('&')

  const signingKey = `${encodeURIComponent(process.env.NETSUITE_CONSUMER_SECRET)}&${encodeURIComponent(process.env.NETSUITE_TOKEN_SECRET)}`
  const signature = crypto.createHmac('sha256', signingKey).update(baseString).digest('base64')

  params.oauth_signature = signature
  params.realm = process.env.NETSUITE_ACCOUNT_ID

  return 'OAuth ' + Object.keys(params).map(k => `${k}="${encodeURIComponent(params[k])}"`).join(', ')
}

export async function fetchNetSuiteInventory() {
  const accountId = process.env.NETSUITE_ACCOUNT_ID
  const url = `https://${accountId}.suitetalk.api.netsuite.com/services/rest/query/v1/suiteql`

  const query = `
    SELECT
      item.id AS itemId,
      item.displayName AS productName,
      item.itemType AS productType,
      item.custitem_smc_model AS model,
      item.custitem_smc_year AS year,
      item.custitem_smc_condition AS condition,
      item.salesprice AS price,
      item.custitem_smc_status AS status
    FROM item
    WHERE item.isinactive = 'F'
      AND item.custitem_smc_lift_inventory = 'T'
    ORDER BY item.id DESC
  `

  const response = await axios.post(
    url,
    { q: query },
    {
      headers: {
        Authorization: generateOAuthHeader('POST', url),
        'Content-Type': 'application/json',
        Prefer: 'transient',
      },
    }
  )

  return response.data.items || []
}
```

Note: The field names custitem_smc_model, custitem_smc_year, custitem_smc_condition, custitem_smc_status, and custitem_smc_lift_inventory are placeholders.
